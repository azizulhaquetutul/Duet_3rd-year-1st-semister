org 100h

.data
result dw ?       ; Variable to store the result

.code
main proc
    mov ax, 16    ; Example input value (you can change this; must be a multiple of 8)

    ; Divide AX by 8 using a right shift
    shr ax, 3      ; AX = AX / 8

    ; Store the result in memory
    mov result, ax ; Save AX (result) in the variable

    ; Program Termination
    mov ah, 4Ch    ; Terminate program
    int 21h
main endp
