org 100h

.data
result dw ?       ; Variable to store the result

.code
main proc
    mov ax, 5     ; Example input value (you can change this)

    ; Step 1: Multiply AX by 32 (2^5)
    mov bx, ax     ; Copy AX to BX
    shl bx, 5      ; BX = AX * 32

    ; Step 2: Multiply AX by 2 (2^1)
    mov cx, ax     ; Copy AX to CX
    shl cx, 1      ; CX = AX * 2

    ; Step 3: Add all components to calculate AX * 35
    add bx, cx     ; BX = AX * 32 + AX * 2
    add bx, ax     ; BX = AX * 32 + AX * 2 + AX (AX * 35)

    ; Store the result in memory
    mov result, bx ; Save BX (result) in the variable

    ; Program Termination
    mov ah, 4Ch    ; Terminate program
    int 21h
main endp
