    mov bx, ax

    ; Step 2: Multiply AX by 8 (2^3)
    mov cx, ax

    ; Step 3: Multiply AX by 2 (2^1)
    mov dx, ax

    ; Step 4: Add results to compute AX * 27
    add bx, cx
