org 100h

.data
result dw ?       ; Variable to store the result

.code
main proc
    mov ax, 64    ; Example input value (you can change this; must be a multiple of 16)

    ; Divide AX by 16 using a right shift
    shr ax, 4      ; AX = AX / 16

    ; Store the result in memory
    mov result, ax ; Save AX (result) in the variable

    ; Program Termination
    mov ah, 4Ch    ; Terminate program
    int 21h
main endp
