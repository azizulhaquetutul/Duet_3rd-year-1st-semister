org 0100h
.data
    msg db "Enter the number of terms (1-9): $"
    n db 1               
    terms db ?           
    result db ?          

.code 
main proc
    
     mov ax,@data
     mov ds,ax
     
     mov ah,9
     lea dx,msg
     int 21h  
     
     mov ah,1
     int 21h
     sub al,'0'
     mov terms,al
     
     xor bh,bh
     mov cl,terms
     
     sum:
     add bh,n
     inc n
     loop sum
     
     mov result,bh
     

    ; Terminate the program
    mov ah, 4ch
    int 21h
main endp
end main
